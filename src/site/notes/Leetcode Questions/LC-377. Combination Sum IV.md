```python
def combinationSum4(self, nums: List[int], target: int) -> int:
	dp = [0]*(target+1)
	nums.sort()
	dp[0] = 1
	for i in range(target+1):
		for c in nums:
			if i >= c:
				dp[i] += dp[i-c]
			else:
				break
	return dp[target]
```
